def energia
  val = @alimentos.collect{|x| x.val_ener}
  gr = @gramos.collect {|x| x*0.01}

  val.zip(gr).map{|x,y| x*y}

  val_tot = 0
  val.each {|x| val_tot += x}
  val_tot
end

def huella_carbono
  hc = @alimentos.collect{|x| x.gei}
  gr = @gramos.collect {|x| x*0.01}

  hc.zip(gr).map{|x,y| x*y}

  hc_tot = 0
  hc.each {|x| hc_tot += x}
  hc_tot
end

def indice_ener
  if energia < 670

  elsif energia > 830

  else
    return 2
  end
end
