def porcentaje_g
  g = @gramos.collect {|x| x*0.01}
  al_g = @alimentos.collect {|x| x.kcal_g}
  tot = @alimentos.collect {|x| x.val_ener}

  kcal_total = 0
  gluc_total = 0

  for i in (0...tot.size) do 
    kcal_total += (tot[i] * g[i])
    gluc_total += (al_g[i] * g[i])
  end

  por = (gluc_total*100.0) / kcal_total
end

def to_s
  a = "Nombre: #{@nombre} -> "
  a << "proteinas: #{porcentaje_p.ceil(1)}%, "
  a << "lípidos: #{porcentaje_l.ceil(1)}%, "
  a << "glúcidos: #{porcentaje_g.ceil(1)}%. "
  a << "Porcentajes aproximados."
end
